## 📋 八、文案使用指南

### **文案选择逻辑**

```typescript
// 伪代码示例
function selectReminder(duration: number, energy: number): string {
  // 1. 根据时长选择文案类别
  let category: string;
  if (duration < 30) return null; // 不提醒
  else if (duration < 60) category = 'gentle'; // 温柔
  else if (duration < 90) category = 'friendly'; // 友善
  else if (duration < 120) category = 'supervision_mild'; // 温和监督
  else if (duration < 180) category = 'supervision_clear'; // 明确监督
  else category = 'supervision_strong'; // 强监督
  
  // 2. 从对应类别随机选择一条文案
  const messages = messageBank[category];
  return messages[Math.floor(Math.random() * messages.length)];
}
```
